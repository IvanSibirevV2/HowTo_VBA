dim i, arr_name(5), Arr_Filter, List

Arr_name(0) = "ПонедельниК"
Arr_name(1) = "Вторник"
Arr_name(2) = "Среда"
Arr_name(3) = "Четверг"
Arr_name(4) = "Пятница"
Arr_name(5) = "Суббота"

Arr_Filter = Filter(Arr_name, "ик", false, 1)

i = 0

Do Until i = UBound(Arr_Filter) + 1
List = List & Arr_Filter(i) & vbCrlf
i = i+1
Loop

MsgBox List
